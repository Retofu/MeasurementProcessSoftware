== Задание

. Дописать программу, которая будет переключать все светодиоды с периодом 0,5 секунды. По нажатию кнопки увеличивать период моргания на 0,1 секунды. При достижении периода 1 секунда сбрасывать до 0,5 секунды. Для задания периода использовать таймер TIM2 (для данного варианта), а прерывания для кнопки сделать через TIM5.

using tMsec = std::uint32_t;
const std::uint32_t SystemCoreClock = 16'000'000U;
tMsec period = 500U;

extern "C"
{
  int __low_level_init(void)
  {
    RCC::APB1ENR::TIM2EN::Enable::Set();
    RCC::APB1ENR::TIM5EN::Enable::Set();
    NVIC::ISER0::Write(1 << 28U); // разрешили глобальное прерывание для TIM2
    NVIC::ISER1::Write(1 << 18U); // разрешили глобальное прерывание для TIM5
    
    const auto TimerDevider = SystemCoreClock / 1'000U;
      
    TIM2::PSC::Write(TimerDevider);
    TIM2::CR1::URS::OverflowEvent::Set();
    TIM2::ARR::Write(period);
    TIM2::SR::UIF::Set(0);
    TIM2::CNT::Write(0);
    TIM2::DIER::UIE::Enable::Set(); // разрешили прерывание по переполнению
    TIM2::CR1::CEN::Enable::Set();
    
    TIM5::PSC::Write(TimerDevider);
    TIM5::CR1::URS::Value1::Set();
    TIM5::ARR::Write(100U);
    TIM5::SR::UIF::Set(0);
    TIM5::CNT::Write(0);
    TIM5::DIER::UIE::Value1::Set(); // разрешили прерывание по переполнению
    TIM5::CR1::CEN::Enable::Set();
    return 1;
  }
}

Pin<GPIOA,5> pin1 ;
Pin<GPIOC,9> pin2 ;

  GPIOC::MODER::MODER13::Input::Set() ;

  RCC::AHB1ENR::GPIOAEN::Enable::Set() ;

  GPIOC::MODER::MODER5::Output::Set() ;
  GPIOC::MODER::MODER8::Output::Set() ;
  GPIOC::MODER::MODER9::Output::Set() ;
  
  for(;;)
  {
    if(allflickMode.GetTimerFlag())
    {
      allflickMode.Update();
    }
    if(button.GetTimerFlag())
    {
      period = (period <= tMsec(1000U)) ? (period + tMsec(100U)) : tMsec(500U);
    }
  }
  
  return 1;

  static void handler()
  {
    if(TIM2::SR::UIF::InterruptPending::IsSet() and TIM2::DIER::UIE::Enable::IsSet())

      allflickMode.SetTimerFlag(true);

#ifndef BUTTON
#define BUTTON

#include <ibutton.h> // for IButton

template<typename Port, int pinNumber>
class Button : public IButton
{
public:
  bool IsPressed() override
  {
    if(((Port::IDR::Get() & (1 << pinNumber)) == 0) && (isSet))
    {
      isSet = false;
    } else if(((Port::IDR::Get() & (1 << pinNumber)) == (1 << pinNumber)) && (!isSet))
    {
      isSet = true;
      return true;
    }
  }
  bool GetTimerFlag()
  {
    return TimerFlag;
  }
  bool SetTimerFlag(bool flag)
  {
    TimerFlag = flag;
  }
private:
  bool TimerFlag = false;
  bool isSet = true;
};

#endif

#ifndef ALLFLICKMODE
#define ALLFLICKMODE

#include "imode.h" // for IMode
#include <array>

template<auto&...pins>
class AllFlickMode : public IMode
{
public:
  void Update() override
  {
    for(std::size_t i = 0; i < leds.size(); i++)
    {
      leds[i]->Toggle();
    }
  }
  void SetToDefault() override
  {
    for(auto it: leds)
    {
      it->Off();
    }
  }
  bool GetTimerFlag()
  {
    return TimerFlag;
  }
  bool SetTimerFlag(bool flag)
  {
    TimerFlag = flag;
  }
private:
  bool TimerFlag = false;
  std::array<IPin*, sizeof ...(pins)> leds = {&pins...};
};

#endif
